import { useState } from 'react'

export const MyDynamicButton = () => {
  const [count, setCount] = useState(0)
  return (
      <div>
        <button onClick={() => setCount(count + 1)}>Clicked {count} times!</button>
        <h3>{count}</h3>
        <div class="tip" markdown="1">Have **fun!**</div>
      </div>
    
  )
}
